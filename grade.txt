#include <stdio.h>
#include <stdlib.h>
#include <time.h>
#define M 3
#define N 5

static float ave_s[M],ave_c[N];//定义全局数组，学生平均成绩和课程平均成绩
static int x,y;//定义全局变量，最高分对应的学生和课程

void average_s(float array[M][N])//计算学生平均分的函数
{
    int i,j;
    float sum_s;
    for(i=0;i<M;i++)
    {
        sum_s=0;
        for(j=0;j<N;j++)
        {
            sum_s=sum_s+array[i][j];
            ave_s[i]=sum_s/N;
        }
    }
}

void average_c(float array[M][N])//计算课程平均分的函数
{
    int i,j;
    float sum_c;
    for(j=0;j<N;j++)
    {
        sum_c=0;
        for(i=0;i<M;i++)
        {
            sum_c=sum_c+array[i][j];
            ave_c[j]=sum_c/M;
        }
    }
}

void high(float array[][N])//找出最高分对应的学生和课程
{
    int i,j;
    float Max=array[0][0];
    for(i=0;i<M;i++)
    {
        for(j=0;j<N;j++)
        {
            if(Max<array[i][j])
            {
                Max=array[i][j];
                x=i;
                y=j;
            }
        }
    }
}

float s()//计算方差的函数
{
    int i;
    float sum1=ave_s[0],sum2=ave_s[0]*ave_s[0];
    for(i=1;i<M;i++)
    {
        sum1=sum1+ave_s[i];
        sum2=sum2+ave_s[i]*ave_s[i];
    }
    return (sum2/M-(sum1/M)*(sum1/M));
}

void initial(float array[][N])//定义随机数组的函数
{
    int i,j;
    srand((unsigned)time(NULL));//将随机数函数的种子设为时间
    for(i=0;i<M;i++)
    {
        for(j=0;j<N;j++)
        {
            array[i][j]=60+rand()%40;//将生成的随机数设置到60-99之间，并赋值给成绩数组
        }
    }
}
int main()
{
    int i;
    float s2;
    float score[M][N]={{70,70,70,70,70},{70,70,70,70,70},{70,70,70,70,70}};//定义成绩数组
    //float score[M][N]={{70,70,70,70,70},{70,70,70,100,70},{70,70,70,70,70}};
    /*
    float score[M][N];
    initial(score);
    */
    printf("Averages of students are ");
    average_s(score);//调用函数计算学生平均分
    for(i=0;i<3;i++)
    {
        printf("%lf ",ave_s[i]);
    }
    printf("\nAverages of courses are ");
    average_c(score);//调用函数计算课程平均分
    for(i=0;i<5;i++)
    {
        printf("%lf ",ave_c[i]);
    }
    high(score);//调用函数计算最高分对应的学生和课程
    printf("\nThe score of the No.%d course of the No.%d student is the highest.",y+1,x+1);
    s2=s();//调用函数计算方差
    printf("\nThe s2 =%lf",s2);
    return 0;
}
